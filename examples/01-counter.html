react equivalent

function Counter() {
  const [n, setN] = useState(0);
  return (
    <div style={{display:'flex',flexDirection:'column',gap:8}}>
      <span>{n}</span>
      <div style={{display:'flex',flexDirection:'row',gap:8}}>
        <button onClick={()=>setN(n=>n-1)}>-</button>
        <button onClick={()=>setN(n=>n+1)}>+</button>
      </div>
    </div>
  );
}
