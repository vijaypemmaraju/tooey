function Counter() {
  const [n, setN] = useState(0);
  return (
    <div style={{display:'flex',
      flexDirection:'column',gap:8}}>
      <span>{n}</span>
      <div style={{display:'flex',
        flexDirection:'row',gap:8}}>
        <button onClick={()=>setN(n=>n-1)}>
          -
        </button>

        </button>

}

design principles

mnemonic, not numeric

single-letter codes that are guessable: V=vertical, H=horizontal, B=button, T=text

positional arguments

common patterns use position, not keys: [type, content, props] is always the order

short property names

g=gap, p=padding, c=click, v=value, w=width, h=height

symbolic operations

state ops use symbols: + increment, - decrement, ! set, ~ toggle

llm-debuggable

unlike pure numeric encoding, tooey remains readable and modifiable by llms

zero dependencies

tiny runtime with signals-based reactivity. no virtual dom overhead.

component reference

// component types
V = vstack (vertical flex)    H = hstack (horizontal flex)
D = div (container)           G = grid
T = text (span)               B = button
I = input                     S = select
C = checkbox                  L = link
M = image

// common props
g = gap          p = padding       m = margin
w = width        h = height        r = border-radius
bg = background  fg = color        s = style object
c = onclick      x = onchange      v = value
ph = placeholder t = type          d = disabled

// state operations
["stateName", "+"]       // increment
["stateName", "-"]       // decrement
["stateName", "!", val]  // set to val
["stateName", "~"]       // toggle boolean
["stateName", "<", val]  // append to array
["stateName", ">", val]  // prepend to array
["stateName", "X", idx]  // remove from array

// state references
{$: "stateName"}         // reference state value in content
